#Write a Python program to compute the eigenvalues and right eigenvectors of a given square array.

import numpy as np m = np.mat("3 -2;1 0") print("Original matrix:") print("a\n", m) w, v = np.linalg.eig(m) print( "Eigenvalues of the said matrix",w) print( "Eigenvectors of the said matrix",v)

#Write a Python program to compute the factor of a given array by Singular Value Decomposition

import numpy as np a = np.array([[1, 0, 0, 0, 2], [0, 0, 3, 0, 0], [0, 0, 0, 0, 0], [0, 2, 0, 0, 0]], dtype=np.float32) print("Original array:") print(a) U, s, V = np.linalg.svd(a, full_matrices=False) q, r = np.linalg.qr(a) print("Factor of a given array by Singular Value Decomposition:") print("U=\n", U, "\ns=\n", s, "\nV=\n", V)
